  p_user_id UUID,
  p_year INTEGER,
  p_month INTEGER
)

  total_income BIGINT,
  total_expense BIGINT,
  net_amount BIGINT,
  expense_by_category JSONB

DECLARE
  v_start_date DATE;
  v_end_date DATE;
BEGIN
  v_start_date := make_date(p_year, p_month, 1);
  v_end_date := v_start_date + INTERVAL '1 month' - INTERVAL '1 day';
  
  RETURN QUERY
  WITH monthly_transactions AS (
    SELECT t.*, a_debit.type as debit_type, a_credit.type as credit_type,
           a_debit.category as debit_category
    FROM transactions t
    JOIN accounts a_debit ON t.debit_account_id = a_debit.id
    JOIN accounts a_credit ON t.credit_account_id = a_credit.id
    WHERE t.user_id = p_user_id
      AND t.date BETWEEN v_start_date AND v_end_date
      AND t.deleted_at IS NULL
  )
  SELECT
    COALESCE(SUM(CASE WHEN credit_type = 'income' THEN amount ELSE 0 END), 0) as total_income,
    COALESCE(SUM(CASE WHEN debit_type = 'expense' THEN amount ELSE 0 END), 0) as total_expense,
    COALESCE(SUM(CASE WHEN credit_type = 'income' THEN amount ELSE 0 END), 0) -
    COALESCE(SUM(CASE WHEN debit_type = 'expense' THEN amount ELSE 0 END), 0) as net_amount,
    (
      SELECT jsonb_object_agg(category, category_sum)
      FROM (
        SELECT debit_category as category, SUM(amount) as category_sum
        FROM monthly_transactions
        WHERE debit_type = 'expense'
        GROUP BY debit_category
      ) sub
    ) as expense_by_category;
END;
